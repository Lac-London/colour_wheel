// 5. Pack the 32 bytes into a 32-bit integer

uint32_t bitmask = _mm256_movemask_epi8(
    final_mask
);


// 6. The "Fast Reject" Logic

if (bitmask == 0xFFFFFFFF) {
    // ALL 32 pixels are pure green.
    // Skip Neural Network!
    Hardcode_Alpha(0.0f);
} else if (bitmask == 0x00000000) {
    // NONE are green (e.g. solid person).
    // Skip Neural Network!
    Hardcode_Alpha(1.0f);
} else {
    // Mixed 1s and 0s. This is an EDGE.
    // MUST send to TensorRT!
    Evaluate_With_Neural_Network();
}
