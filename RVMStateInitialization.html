std::vector<float> r1i(1*16 * 540*960,   0.0f);

// 2. Create Ort Memory Info

    OrtDeviceAllocator, OrtMemTypeCPU
);

// 3. Wrap C++ pointers into ONNX Tensors

Ort::Value t1 = Ort::Value::CreateTensor<float>(
    mem_info, r1i.data(), r1i.size(), ...);
Ort::Value t2 = Ort::Value::CreateTensor<float>(
    mem_info, r2i.data(), r2i.size(), ...);

    mem_info, r3i.data(), r3i.size(), ...);
